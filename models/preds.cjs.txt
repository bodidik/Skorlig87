"use strict";
const { getDb } = require("../lib/db.cjs");

async function savePred(doc){
  const db = await getDb();
  const col = db.collection("preds");
  doc.at = doc.at ? new Date(doc.at) : new Date();
  await col.insertOne(doc);
  return true;
}

async function getMyLatestPred(fixtureId, userId){
  const db = await getDb();
  const col = db.collection("preds");
  return await col.find({ fixtureId:String(fixtureId), $or:[{userId:String(userId)},{user:String(userId)}] })
                  .sort({ at:-1 }).limit(1).next();
}

async function getPredsForFixture(fixtureId){
  const db = await getDb();
  const col = db.collection("preds");
  return await col.find({ fixtureId:String(fixtureId) }).toArray();
}

module.exports = { savePred, getMyLatestPred, getPredsForFixture };
